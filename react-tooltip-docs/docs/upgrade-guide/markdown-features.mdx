import { Tooltip } from 'react-tooltip'
import 'react-tooltip/dist/react-tooltip.css'

export const TooltipAnchor = ({ children, id, ...rest }) => {
  return (
    <span
      id={id}
      style={{
        display: 'flex',
        justifyContent: 'center',
        alignItems: 'center',
        width: '60px',
        height: '60px',
        borderRadius: '60px',
        color: '#222',
        background: 'rgba(255, 255, 255, 1)',
        cursor: 'pointer',
        boxShadow: '3px 4px 3px rgba(0, 0, 0, 0.5)',
        border: '1px solid #333',
      }}
      onClick={() => {
        alert(`clicked on Tooltip Anchor: ${id}`)
      }}
      {...rest}
    >
      {children}
    </span>
  )
}

# Basic example

## Before V5

```jsx
import ReactTooltip from 'react-tooltip';

<a data-tip="React-tooltip"> ◕‿‿◕ </a>

<ReactTooltip place="top" type="dark" effect="float"/>
```

(v4) `data-tip` = `data-content` (v5)

## V5 - props option

```jsx
import { Tooltip } from 'react-tooltip';
import 'react-tooltip/dist/react-tooltip.css'

<a id="tooltip-anchor-props"> ◕‿‿◕ </a>

<Tooltip anchorId="tooltip-anchor-props" content="hello world!" />
```

<TooltipAnchor id="tooltip-anchor-props">◕‿‿◕</TooltipAnchor>
<Tooltip anchorId="tooltip-anchor-props" content="hello world!" />

## V5 - data attributes

```jsx
import { Tooltip } from 'react-tooltip';
import 'react-tooltip/dist/react-tooltip.css';

<a id="tooltip-anchor-attributes" data-content="hello world!"> ◕‿‿◕ </a>

<Tooltip anchorId="tooltip-anchor-attributes" />
```

<TooltipAnchor id="tooltip-anchor-attributes" data-content="hello world!">
  ◕‿‿◕
</TooltipAnchor>
<Tooltip anchorId="tooltip-anchor-attributes" />
